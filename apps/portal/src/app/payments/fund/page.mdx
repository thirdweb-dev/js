pragma solidity ^0.8.20;

import "@openzeppelin/contracts/token/ERC20/ERC20.sol";
import "@openzeppelin/contracts/access/Ownable.sol";

contract GRAP is ERC20, Ownable {
    constructor(
        string memory name,
        string memory symbol,
        uint256 totalSupply,
        address 0x7DB32D8658B5042237476b4a9C242073F4174A52
    ) ERC20(name, symbol) Ownable(initialOwner) {
        _mint(initialOwner, totalSupply * 1000000000 decimals());
        // NOTE: hapus baris di bawah kalau masih ingin ada owner
    }

    function _update(address from, address to, uint256 amount) internal override {
        require(to != address(0), "LEXXA: transfer to zero address"); //
        super._update(from, to, amount);
    }

    // Opsional: override decimals
    // function decimals() public pure override returns (uint8) {

    // }
}
