import {
  createVaultClient,
  ping,
  createEoa,
  listEoas,
  parseTransaction,
  signTransaction,
  createAccessToken,
} from "@thirdweb-dev/vault-sdk";
import "dotenv/config";

// --- CONFIGURACIÃ“N ---

const SECRET_KEY = process.env.VAULT_SECRET_KEY!;
const SIG_TOKEN = process.env.VAULT_SIG_TOKEN!;
const BASE_TOKEN = process.env.VAULT_BASE_TOKEN!;

const TARGET_ADDRESS = "0x6561fF4034De957fAFeA20b3aCC5E6E9FEb21aCE";

async function main() {
  // 1. Crear cliente del Vault
  const client = await createVaultClient({ secretKey: SECRET_KEY });

  // 2. Hacer ping al Vault (opcional)
  const pingResult = await ping({ client, request: { message: "pong?" } });
  console.log("Ping response:", pingResult);

  // 3. Crear un EOA con metadata que incluya la direcciÃ³n
  const eoa = await createEoa({
    client,
    request: {
      metadata: {
        label: "Wallet externa",
        externalAddress: TARGET_ADDRESS,
      },
    },
  });
  console.log("Nuevo EOA creado (metadatos):", eoa);

  // 4. Crear una transacciÃ³n ficticia desde la direcciÃ³n
  const tx = parseTransaction({
    to: "0xAbC123456789abcdef123456789abcdef1234567", // direcciÃ³n de destino ficticia
    value: 0n,
    chainId: 1,
    maxFeePerGas: 30n * 10n ** 9n,
    maxPriorityFeePerGas: 1n * 10n ** 9n,
    gasLimit: 21_000,
  });

  // 5. Firmar transacciÃ³n (usando la direcciÃ³n como `from`)
  const signedTx = await signTransaction({
    client,
    request: {
      auth: { accessToken: SIG_TOKEN },
      options: {
        from: TARGET_ADDRESS,
        transaction: tx,
      },
    },
  });
  console.log("TransacciÃ³n firmada:", signedTx);

  // 6. Crear un access token vinculado a la direcciÃ³n
  const token = await createAccessToken({
    client,
    request: {
      metadata: {
        name: "Token para wallet especÃ­fica",
        wallet: TARGET_ADDRESS,
      },
      policies: [
        {
          type: "eoa:signMessage",
          chainId: 1,
          eoa: TARGET_ADDRESS,
        },
      ],
    },
  });
  console.log("Access token creado:", token);

  // 7. Listar EOAs que tengan la direcciÃ³n como metadato
  const matchingEoas = await listEoas({
    client,
    request: {
      metadata: {
        externalAddress: TARGET_ADDRESS,
      },
    },
  });
  console.log("EOAs encontrados con esa direcciÃ³n:", matchingEoas);
}

// Ejecutar
main().catch((err) => {
  console.error("Error durante la ejecuciÃ³n:", err);
});
